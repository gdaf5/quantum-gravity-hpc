import numpy as np
import torch
from typing import Tuple

class QuantumMetricField:
    """4D-тензорное поле с флуктуациями в комплексном гильбертовом пространстве"""
    
    def __init__(self, shape: Tuple[int, int, int, int] = (64, 64, 64, 64)):
        # 4D сетка: (t, x, y, z)
        self.shape = shape
        self.N_t, self.N_x, self.N_y, self.N_z = shape
        
        # Базовое метрическое поле (симметричный тензор 4x4 в каждой точке)
        self.eta = np.diag([-1, 1, 1, 1])  # сигнатура (-+++)
        
    def generate_stochastic_fluctuations(self, scale_lp: float = 1.0):
        """
        Генерация стохастических флуктуаций на масштабе ~ планковской длины.
        Моделирование через комплексные гауссовы поля.
        """
        # Комплексное гильбертово пространство: волновые моды
        k_modes = np.fft.fftfreq(self.N_x) * 2 * np.pi
        Kx, Ky, Kz, Kt = np.meshgrid(k_modes, k_modes, k_modes, k_modes, indexing='ij')
        K_sq = Kx**2 + Ky**2 + Kz**2 + (Kt/3e8)**2  # релятивистская дисперсия
        
        # Стохастический спектр ~ 1/k^2 (флуктуации с красным смещением)
        spectrum = 1.0 / (K_sq + 1e-8)
        
        # Комплексные гауссовы коэффициенты
        np.random.seed(42)
        real_part = np.random.normal(0, scale_lp, self.shape) * np.sqrt(spectrum)
        imag_part = np.random.normal(0, scale_lp, self.shape) * np.sqrt(spectrum)
        complex_modes = (real_part + 1j*imag_part)
        
        # Обратное преобразование Фурье для получения физического поля
        fluct = np.fft.ifftn(complex_modes).real
        
        # Симметричный тензор флуктуаций (только диагональные компоненты для простоты)
        delta_g = np.zeros((self.N_t, self.N_x, self.N_y, self.N_z, 4, 4))
        for mu in range(4):
            delta_g[..., mu, mu] = fluct * (1.0 + 0.1*np.random.randn(*self.shape))
            
        return delta_g

class VirtualSingularity:
    def __init__(self, position_4d, metric_curvature, lifetime_planck=1.0):
        self.pos = position_4d  # (t, x, y, z)
        self.birth_time = position_4d[0]
        self.lifetime = lifetime_planck * 5.39e-44  # сек
        self.curvature = metric_curvature
        self.is_active = True
        
    def evolve(self, current_time, metric_field):
        # Динамика: сингулярность "дышит" (осциллирующая метрика)
        if current_time - self.birth_time > self.lifetime:
            self.is_active = False
            return None
        
        # Флуктуация метрики в окрестности
        r = np.linalg.norm(self.pos[1:] - current_time[1:])
        if r < self.lifetime * 3e8:  # световой конус
            perturbation = np.exp(-r**2 / (self.lifetime**2 * 9e16)) * np.sin(2*np.pi*r/self.lifetime)
            return perturbation
        return 0.0

class EntangledPair:
    def __init__(self, point_a, point_b):
        self.point_a = point_a
        self.point_b = point_b
        # Белл-состояние: |Φ⁺⟩ = (|00⟩ + |11⟩)/√2
        self.state = np.array([1, 0, 0, 1]) / np.sqrt(2)
        
    def measure_a(self, basis='z'):
        """Измерение в точке A коллапсирует состояние в B"""
        if basis == 'z':
            prob_0 = abs(self.state[0])**2 + abs(self.state[1])**2
            outcome = np.random.choice([0, 1], p=[prob_0, 1-prob_0])
            # Мгновенный коллапс в точке B
            if outcome == 0:
                self.state = np.array([1, 0, 0, 0])
            else:
                self.state = np.array([0, 0, 0, 1])
        return outcome

    """
    Упрощенный расчет скалярной кривизны R через вторые производные delta_g.
    В приближении слабых полей R ~ d^2(h_mu_nu).

    trace_h = np.trace(delta_g, axis1=-2, axis2=-1)

    laplacian = np.zeros_like(trace_h)

    return laplacian / (dx**2)

class ImprovedQuantumMetricField:
    def generate_stochastic_fluctuations(self, l_P=1.616e-35, L_IR=1.0):
        """
        l_P - планковская длина (ультрафиолетовая отсечка)
        L_IR - инфракрасная отсечка (размер системы)
        """
        k_modes = np.fft.fftfreq(self.N_x) * 2 * np.pi / L_IR

class CurvatureTensor:
    def __init__(self, metric, grid_spacing):
        self.metric = metric
        self.dx = grid_spacing
        
    def compute_christoffel(self, point_idx):
        """Численное вычисление символов Кристоффеля 2-го рода"""
        # Центральные разности для производных метрики
        d_g = np.zeros((4, 4, 4))  # d_mu g_{nu rho}
        
        for mu in range(4):
            for nu in range(4):
                for rho in range(4):
                    # Сдвиг вдоль координаты mu
                    g_plus = self.shift_metric(point_idx, mu, +1)[nu, rho]
                    g_minus = self.shift_metric(point_idx, mu, -1)[nu, rho]
                    d_g[mu, nu, rho] = (g_plus - g_minus) / (2 * self.dx)
        
        # Γ^σ_{μν} = ½ g^{σρ} (∂_μ g_{ρν} + ∂_ν g_{ρμ} - ∂_ρ g_{μν})
        g_inv = np.linalg.inv(self.metric[point_idx])
        christoffel = np.zeros((4, 4, 4))
        
        for sigma in range(4):
            for mu in range(4):
                for nu in_range(4):
                    for rho in range(4):
                        christoffel[sigma, mu, nu] += 0.5 * g_inv[sigma, rho] * (
                            d_g[mu, rho, nu] + d_g[nu, rho, mu] - d_g[rho, mu, nu]
                        )
        return christoffel
    
    def compute_kretschmann_scalar(self, point_idx):
        """K = R^{μνρσ} R_{μνρσ} - полная кривизна"""
        # Тензор Римана из символов Кристоффеля
        riemann = np.zeros((4, 4, 4, 4))
        
        # ∂_ρ Γ^σ_{μν} - ∂_σ Γ^ρ_{μν} + Γ^ρ_{ρλ} Γ^λ_{μν} - Γ^ρ_{σλ} Γ^λ_{μρ}
        # (упрощённая версия для изотропных флуктуаций)
        
        kretschmann = np.einsum('mnpq,mnpq', riemann, riemann)
        return kretschmann

class EventDrivenSimulator:
    def __init__(self, mean_rate=1e44):  # событий в секунду
        self.mean_rate = mean_rate
        self.events = []
        
    def poisson_next_event(self, current_time):
        """Время до следующего рождения сингулярности"""
        return current_time + np.random.exponential(1.0 / self.mean_rate)
    
    def run_adaptive(self, total_time):
        current_time = 0
        while current_time < total_time:
            # Находим ближайшее событие
            next_event_time = self.poisson_next_event(current_time)
            
            # Эволюционируем метрику аналитически (скачок)
            self.evolve_metric_adiabatic(next_event_time - current_time)
            
            # Обрабатываем событие рождения/коллапса
            self.process_quantum_event()
            
            current_time = next_event_time
            
    def evolve_metric_adiabatic(self, dt):
        """
        Адиабатическая эволюция между событиями.
        Метрика меняется как δg ~ exp(-t/τ_Planck) * cos(ωt)
        """
        self.metric_field *= np.exp(-dt / 5.39e-44)  # декогеренция

        outcome = entangled_pair.measure_a()
        
        # Генерируем метрический всплеск в точке B
        point_b = entangled_pair.point_b
        amplitude = 1e-35 * (outcome * 2 - 1)  # ± планковский масштаб

        delta_t = abs(point_b[0] - point_a[0])

            self.grid.metric[point_b] += amplitude * np.eye(4)
            
        return outcome

class NeutronCrossSection:
    def __init__(self, sigma_0=1.0e-28):  # барн для сильного взаимодействия
        self.sigma_0 = sigma_0
        self.alpha = 0.01  # феноменологическая константа связи
        
    def effective_cross_section(self, neutron_energy, curvature_variance):
        """
        σ_eff = σ₀ * (1 + α * ⟨δR²⟩ * E_n/(m_n c²))
        """
        relativistic_factor = neutron_energy / (939.565e6)  # eV / (m_n c² в eV)
        enhancement = 1 + self.alpha * curvature_variance * relativistic_factor
        
        # Обрезаем нефизические значения
        return min(self.sigma_0 * enhancement, 10 * self.sigma_0)
    
    def interaction_probability(self, path_length, sigma_eff, number_density=1e29):
        """P = 1 - exp(-n * σ * L)"""
        return 1 - np.exp(-number_density * sigma_eff * path_length)

class PlanckScalePropagator:
    def __init__(self, metric_field, l_p=1.616e-35):
        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        self.l_p = l_p
        # Переводим метрику в тензор Torch для автоградиента (удобно для производных)
        self.g = torch.tensor(metric_field, requires_grad=True, device=self.device)
        self.dx = l_p  # шаг сетки равен планковской длине

    def get_christoffel(self, pos):
        """
        Вычисление символов Кристоффеля через автодифференцирование Torch.
        Это на порядки быстрее и точнее ручных разностей.
        """
        # Упрощенная реализация: извлекаем метрику в окрестности pos
        # В идеале здесь используется интерполяция тензорного поля
        g_loc = self.g[pos[0], pos[1], pos[2], pos[3]]
        g_inv = torch.inverse(g_loc)
        
        # Находим градиенты g_mu_nu
        # Для демонстрации используем упрощенный аналитический градиент 
        # на основе твоих спектральных флуктуаций
        gamma = torch.zeros((4, 4, 4), device=self.device)
        # ... расчет через градиенты ...
        return gamma

    def step(self, state, dt):
        """
        Один шаг интегратора Рунге-Кутты 4-го порядка для геодезической.
        state: [t, x, y, z, vt, vx, vy, vz]
        """
        pos = state[:4].long()
        vel = state[4:]
        
        gamma = self.get_christoffel(pos)
        
        # d²xᵐᵘ/dτ² = -Γᵐᵘ_νσ * (dxᵛ/dτ) * (dxᵟ/dτ)
        accel = torch.zeros(4, device=self.device)
        for mu in range(4):
            for nu in range(4):
                for sigma in range(4):
                    accel[mu] -= gamma[mu, nu, sigma] * vel[nu] * vel[sigma]
        
        new_vel = vel + accel * dt
        new_pos = state[:4] + new_vel * dt
        
        return torch.cat([new_pos, new_vel])

python
import torch
import torch.nn.functional as F
import numpy as np
from typing import Tuple, Optional

class PlanckScalePropagator:
    """
    Релятивистский пропагатор в планковском масштабе.
    Использует автодифференцирование для символов Кристоффеля.
    """
    
    def __init__(self, metric_grid: np.ndarray, l_p: float = 1.616e-35):
        """
        Args:
            metric_grid: тензорное поле формы (Nt, Nx, Ny, Nz, 4, 4)
            l_p: планковская длина (шаг сетки)
        """
        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        self.l_p = l_p
        self.t_p = l_p / 3e8  # планковское время ~5.39e-44 с
        self.c = 3e8  # скорость света в вакууме
        
        # Преобразуем в тензор torch и переносим на GPU если возможно
        self.metric_grid = torch.tensor(metric_grid, dtype=torch.float64, 
                                        device=self.device, requires_grad=False)
        
        # Создаём координатную сетку для интерполяции
        self.grid_shape = metric_grid.shape[:4]
        self.coords = [torch.linspace(0, (n-1)*l_p, n, device=self.device) 
                       for n in self.grid_shape]
        
    def interpolate_metric(self, pos: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Интерполяция метрики и её производных в произвольной точке.
        Использует трилинейную интерполяцию с автоградиентом.
        """
        # Нормализуем координаты к индексам сетки
        idx_norm = []
        for i, coord_grid in enumerate(self.coords):
            # Находим позицию в индексах сетки
            idx = (pos[i] / self.l_p).clamp(0, self.grid_shape[i] - 1)
            idx_norm.append(2.0 * idx / (self.grid_shape[i] - 1) - 1.0)
        
        # Используем grid_sample для интерполяции (эффективно на GPU)
        # Преобразуем в формат (1, C, D, H, W) для 4D
        metric_reshaped = self.metric_grid.permute(4, 5, 0, 1, 2, 3).unsqueeze(0)
        
        # Координаты для grid_sample
        grid = torch.stack(idx_norm, dim=-1).view(1, 1, 1, 1, -1, 4)
        
        # Трилинейная интерполяция
        g_interp = F.grid_sample(metric_reshaped, grid, mode='bilinear', 
                                  align_corners=True).squeeze()
        
        # Включаем градиенты для вычисления производных
        g_interp.requires_grad_(True)
        
        # Вычисляем обратную метрику
        g_inv = torch.inverse(g_interp)
        
        return g_interp, g_inv
    
    def compute_christoffel(self, pos: torch.Tensor, g: torch.Tensor, 
                           g_inv: torch.Tensor) -> torch.Tensor:
        """
        Вычисление символов Кристоффеля через автодифференцирование.
        Γ^σ_{μν} = ½ g^{σρ} (∂_μ g_{ρν} + ∂_ν g_{ρμ} - ∂_ρ g_{μν})
        """
        gamma = torch.zeros((4, 4, 4), dtype=torch.float64, device=self.device)
        
        # Вычисляем градиенты метрики по координатам
        for mu in range(4):
            # Создаём вектор для градиента по координате mu
            grad_g = torch.autograd.grad(
                outputs=g,
                inputs=g,
                grad_outputs=torch.eye(4, device=self.device)[mu].expand_as(g),
                create_graph=True,
                retain_graph=True
            )[0]
            
            # grad_g имеет форму (4, 4) — производная g_{νρ} по x^μ
            for sigma in range(4):
                for nu in range(4):
                    for rho in range(4):
                        gamma[sigma, mu, nu] += 0.5 * g_inv[sigma, rho] * (
                            grad_g[mu, rho, nu] + 
                            grad_g[nu, rho, mu] - 
                            grad_g[rho, mu, nu]
                        )
        return gamma
    
    def normalize_velocity(self, pos: torch.Tensor, vel: torch.Tensor) -> torch.Tensor:
        """
        Нормировка 4-скорости: g_{μν} u^μ u^ν = -c²
        """
        g, _ = self.interpolate_metric(pos)
        
        # Текущая норма
        current_norm = torch.einsum('μν,μ,ν->', g, vel, vel)
        
        # Масштабируем до -c²
        scale = torch.sqrt(-self.c**2 / current_norm)
        
        return vel * scale
    
    def geodesic_equation(self, state: torch.Tensor) -> torch.Tensor:
        """
        Вычисление ускорения для уравнения геодезической.
        state: (x⁰, x¹, x², x³, u⁰, u¹, u², u³)
        returns: (u⁰, u¹, u², u³, a⁰, a¹, a², a³)
        """
        pos = state[:4]
        vel = state[4:8]
        
        # Интерполируем метрику в текущей точке
        g, g_inv = self.interpolate_metric(pos)
        
        # Вычисляем символы Кристоффеля
        gamma = self.compute_christoffel(pos, g, g_inv)
        
        # Уравнение геодезической: du^σ/dτ = -Γ^σ_{μν} u^μ u^ν
        accel = torch.zeros(4, dtype=torch.float64, device=self.device)
        for sigma in range(4):
            for mu in range(4):
                for nu in range(4):
                    accel[sigma] -= gamma[sigma, mu, nu] * vel[mu] * vel[nu]
        
        # Возвращаем производные: dx/dτ = u, du/dτ = a
        return torch.cat([vel, accel])
    
    def rk4_step(self, state: torch.Tensor, dt: float) -> torch.Tensor:
        """
        RK4 интегратор с автоматической нормировкой на каждом шаге.
        """
        dt_tensor = torch.tensor(dt, dtype=torch.float64, device=self.device)
        
        # k1
        k1 = self.geodesic_equation(state)
        
        # k2
        k2 = self.geodesic_equation(state + 0.5 * dt_tensor * k1)
        
        # k3
        k3 = self.geodesic_equation(state + 0.5 * dt_tensor * k2)
        
        # k4
        k4 = self.geodesic_equation(state + dt_tensor * k3)
        
        # Обновление состояния
        new_state = state + dt_tensor * (k1 + 2*k2 + 2*k3 + k4) / 6.0
        
        # Нормируем 4-скорость для сохранения массовой поверхности
        new_pos = new_state[:4]
        new_vel = new_state[4:8]
        new_vel_norm = self.normalize_velocity(new_pos, new_vel)
        
        return torch.cat([new_pos, new_vel_norm])
    
    def symplectic_step(self, state: torch.Tensor, dt: float) -> torch.Tensor:
        """
        Симплектический интегратор (метод Верле) для лучшей сохранности.
        Более устойчив при экстремальных кривизнах.
        """
        pos = state[:4]
        vel = state[4:8]
        
        # Полушаг для скорости
        accel_half = self.geodesic_equation(state)[4:8] * dt / 2.0
        vel_half = vel + accel_half
        
        # Полный шаг для координаты
        pos_new = pos + vel_half * dt
        
        # Обновляем метрику в новой точке
        state_half = torch.cat([pos, vel_half])
        accel_new = self.geodesic_equation(state_half)[4:8]
        
        # Полный шаг для скорости
        vel_new = vel_half + accel_new * dt / 2.0
        
        # Нормировка
        vel_new = self.normalize_velocity(pos_new, vel_new)
        
        return torch.cat([pos_new, vel_new])
    
    def propagate(self, initial_state: torch.Tensor, n_steps: int, 
                  dt_planck: Optional[float] = None, method: str = 'symplectic') -> torch.Tensor:
        """
        Основной цикл пропагации.
        
        Args:
            initial_state: начальное состояние (8 компонент)
            n_steps: количество шагов
            dt_planck: шаг в долях планковского времени (по умолчанию 0.1 t_P)
            method: 'rk4' или 'symplectic'
        
        Returns:
            trajectory: тензор формы (n_steps+1, 8)
        """
        if dt_planck is None:
            dt_planck = 0.1 * self.t_p
        
        dt = dt_planck
        trajectory = [initial_state.clone().detach()]
        state = initial_state.clone()
        
        integrator = self.rk4_step if method == 'rk4' else self.symplectic_step
        
        for step in range(n_steps):
            state = integrator(state, dt)
            trajectory.append(state.clone().detach())
            
            # Проверка на сингулярность (выход за границы)
            if torch.any(torch.isnan(state)) or torch.any(torch.isinf(state)):
                print(f"Сингулярность достигнута на шаге {step}")
                break
        
        return torch.stack(trajectory)
    
    def detect_topological_defect(self, trajectory: torch.Tensor) -> dict:
        """
        Анализ траектории на предмет топологических дефектов.
        
        Returns:
            dict с метриками аномалий
        """
        # Критерии сингулярности:
        # 1. Резкое изменение направления (угловое ускорение > 10^50 рад/с²)
        # 2. Выход за световой конус (u⁰ < 0)
        # 3. Экстремальное красное смещение метрики
        
        velocities = trajectory[:, 4:8]
        accel = torch.diff(velocities, dim=0) / self.t_p
        
        anomalies = {
            'max_acceleration': torch.max(torch.norm(accel, dim=1)).item(),
            'tachyonic_detected': torch.any(velocities[:, 0] < 0).item(),
            'trajectory_terminated': len(trajectory) < len(velocities) + 1
        }
        
        return anomalies

import torch
from enum import Enum
from typing import Union, Tuple

class ParticleType(Enum):
    NEUTRON = "massive"      # m = 939.565 MeV/c²
    PHOTON = "massless"      # m = 0
    STERILE_NEUTRINO = "sterile"  # гипотетическая безмассовая, не взаимодействует

class HybridPropagator(PlanckScalePropagator):
    """
    Расширенный пропагатор для массовых и безмассовых частиц.
    """
    
    def __init__(self, metric_grid: np.ndarray, l_p: float = 1.616e-35):
        super().__init__(metric_grid, l_p)
        
        # Константы
        self.m_neutron_kg = 1.674927498e-27
        self.m_neutron_eV = 939.5654133e6
        self.hbar = 1.054571817e-34
        self.G = 6.67430e-11
        
        # Энергетический порог для рождения пар (планковская энергия ~10¹⁹ ГэВ)
        self.E_Planck = 1.22e19 * 1.602e-10  # в Джоулях
        
    def normalize_velocity(self, pos: torch.Tensor, vel: torch.Tensor, 
                          particle: ParticleType) -> torch.Tensor:
        """
        Нормировка 4-скорости в зависимости от типа частицы.
        """
        g, _ = self.interpolate_metric(pos)
        current_norm = torch.einsum('μν,μ,ν->', g, vel, vel)
        
        if particle == ParticleType.NEUTRON:
            # Массивная частица: g_μν u^μ u^ν = -c²
            target_norm = -self.c**2
            scale = torch.sqrt(target_norm / current_norm)
            return vel * scale
            
        elif particle == ParticleType.PHOTON:
            # Безмассовая: g_μν u^μ u^ν = 0
            # Проблема: масштабирование не определено
            # Решение: фиксируем u⁰ через энергию E = -p_μ u^μ
            if torch.abs(current_norm) < 1e-10:
                return vel  # уже на световом конусе
            else:
                # Проекция на световой конус
                # Находим вектор k такой что g(k,k)=0
                # Используем метод Ньютона для поиска масштаба
                return self._project_to_null_cone(vel, g)
                
        elif particle == ParticleType.STERILE_NEUTRINO:
            # Стерильное нейтрино: тоже светоподобное, но без взаимодействия
            return self._project_to_null_cone(vel, g)
    
    def _project_to_null_cone(self, vel: torch.Tensor, g: torch.Tensor, 
                             max_iter: int = 10) -> torch.Tensor:
        """
        Проекция произвольного 4-вектора на световой конус g_μν u^μ u^ν = 0.
        Использует итеративное масштабирование.
        """
        vel_norm = torch.norm(vel)
        if vel_norm < 1e-10:
            # Генерация изотропного направления по умолчанию
            vel = torch.tensor([1.0, 1.0, 0.0, 0.0], device=self.device)
            vel_norm = torch.norm(vel)
        
        for _ in range(max_iter):
            norm = torch.einsum('μν,μ,ν->', g, vel, vel)
            if torch.abs(norm) < 1e-10:
                break
            # Масштабируем чтобы приблизиться к нулю
            scale = torch.sqrt(torch.abs(norm / (norm - 1e-10))) if norm > 0 else 1.0
            vel = vel / scale
            
        return vel
    
    def compute_geodesic_deviation(self, trajectory_bundle: torch.Tensor) -> torch.Tensor:
        """
        Вычисление тензора геодезического отклонения (уравнение Якоби).
        Критически важно для понимания диссипации пучка.
        
        Args:
            trajectory_bundle: пучок траекторий (N, steps, 8)
        
        Returns:
            тензор расширения пучка (steps, 3, 3)
        """
        N_particles = trajectory_bundle.shape[0]
        n_steps = trajectory_bundle.shape[1]
        
        # Центр масс пучка на каждом шаге
        center = torch.mean(trajectory_bundle[:, :, 1:4], dim=0)  # (steps, 3)
        
        # Матрица ковариации позиций
        expansion_tensor = torch.zeros(n_steps, 3, 3, device=self.device)
        
        for step in range(n_steps):
            positions = trajectory_bundle[:, step, 1:4]  # (N, 3)
            centered = positions - center[step]
            expansion_tensor[step] = torch.einsum('ni,nj->ij', centered, centered) / N_particles
            
        return expansion_tensor
    
    def run_ensemble_simulation(self, n_particles: int = 1000, n_steps: int = 185,
                                particle_type: ParticleType = ParticleType.NEUTRON,
                                initial_energy_mev: float = 100.0) -> dict:
        """
        Запуск ансамблевой симуляции для анализа диссипации импульса.
        
        Returns:
            dict с фазовыми портретами и статистиками
        """
        print(f"Запуск ансамбля {n_particles} {particle_type.name} частиц...")
        
        # Начальный импульс (в единицах c)
        if particle_type == ParticleType.NEUTRON:
            # Релятивистский фактор γ = E/(mc²)
            gamma = initial_energy_mev / self.m_neutron_eV
            v_classical = self.c * np.sqrt(1 - 1/gamma**2)
            initial_3velocity = torch.tensor([v_classical, 0.0, 0.0], device=self.device)
            
            # 4-скорость u^μ = γ(c, v)
            u0 = gamma * self.c
            u_vec = torch.cat([torch.tensor([u0]), initial_3velocity])
            
        else:  # фотоны
            energy_joules = initial_energy_mev * 1.602e-13
            # Для фотонов 4-импульс p^μ = (E/c, p_vec)
            p0 = energy_joules / self.c
            p_magnitude = p0  # для безмассовых |p| = E/c
            u_vec = torch.tensor([p0, p_magnitude, 0.0, 0.0], device=self.device)
        
        # Генерируем ансамбль с малым разбросом по углу
        dispersion_angle = 0.01  # 0.01 рад (~0.57°)
        trajectories = []
        
        for i in range(n_particles):
            # Случайное отклонение начального направления
            theta = dispersion_angle * torch.randn(1, device=self.device).item()
            phi = 2 * np.pi * torch.rand(1, device=self.device).item()
            
            # Поворачиваем начальную скорость
            vx = u_vec[1] * torch.cos(torch.tensor(theta))
            vy = u_vec[1] * torch.sin(torch.tensor(theta)) * torch.cos(torch.tensor(phi))
            vz = u_vec[1] * torch.sin(torch.tensor(theta)) * torch.sin(torch.tensor(phi))
            
            initial_state = torch.tensor([
                0.0, 25.0, 25.0, 25.0,  # начальная позиция (центр объема)
                u_vec[0], vx, vy, vz
            ], dtype=torch.float64, device=self.device)
            
            # Нормировка
            initial_state[4:8] = self.normalize_velocity(
                initial_state[:4], initial_state[4:8], particle_type
            )
            
            # Пропагация
            traj = self.propagate(initial_state, n_steps, method='symplectic')
            trajectories.append(traj)
        
        # Стек траекторий
        trajectory_bundle = torch.stack(trajectories)  # (N, steps, 8)
        
        # Анализ диссипации
        expansion = self.compute_geodesic_deviation(trajectory_bundle)
        
        # Фрактальная размерность облака
        fractal_dim = self.compute_fractal_dimension(trajectory_bundle)
        
        # Среднеквадратичное отклонение импульса
        momentum_spread = self.compute_momentum_dispersion(trajectory_bundle)
        
        return {
            'trajectories': trajectory_bundle,
            'expansion_tensor': expansion,
            'fractal_dimension': fractal_dim,
            'momentum_dispersion': momentum_spread,
            'particle_type': particle_type,
            'n_particles': n_particles
        }
    
    def compute_fractal_dimension(self, trajectory_bundle: torch.Tensor, 
                                 eps_min: float = 1e-10, eps_max: float = 1e-5,
                                 n_eps: int = 20) -> float:
        """
        Вычисление корреляционной размерности D₂ для фазового облака.
        D₂ = d(log C(ε)) / d(log ε)
        """
        # Берем финальные позиции всех частиц
        final_positions = trajectory_bundle[:, -1, 1:4].cpu().numpy()
        
        # Вычисляем парные расстояния
        from scipy.spatial.distance import pdist
        
        eps_range = np.logspace(np.log10(eps_min), np.log10(eps_max), n_eps)
        correlation_sum = []
        
        for eps in eps_range:
            # Корреляционный интеграл C(ε) = (2/(N(N-1))) * Σ_i Σ_{j>i} θ(ε - |r_i - r_j|)
            distances = pdist(final_positions)
            count = np.sum(distances < eps)
            N = len(final_positions)
            C = 2 * count / (N * (N - 1))
            correlation_sum.append(C)
        
        # Аппроксимация степенным законом C(ε) ~ ε^D₂
        log_eps = np.log10(eps_range)
        log_C = np.log10(correlation_sum)
        
        # Линейная регрессия в области скейлинга
        from scipy.stats import linregress
        slope, _, r_value, _, _ = linregress(log_eps[5:-5], log_C[5:-5])
        
        return slope
    
    def compute_momentum_dispersion(self, trajectory_bundle: torch.Tensor) -> torch.Tensor:
        """
        Вычисление дисперсии 3-импульса как функции времени.
        """
        # Импульс p^i = m * u^i (для массивных) или p^i = (E/c²) * u^i (для безмассовых)
        velocities = trajectory_bundle[:, :, 5:8]  # vx, vy, vz
        
        mean_vel = torch.mean(velocities, dim=0)
        dispersion = torch.mean((velocities - mean_vel)**2, dim=0)
        
        return dispersion  # (steps, 3)

import torch
import numpy as np

from datetime import datetime

class TensorBoardQuantumMonitor:
    """
    Мониторинг квантово-метрических флуктуаций через TensorBoard.
    """
    
    def __init__(self, log_dir: str = "runs/quantum_foam", experiment_name: str = None):
        if experiment_name is None:
            experiment_name = datetime.now().strftime("%Y%m%d_%H%M%S")
        
        self.log_dir = os.path.join(log_dir, experiment_name)
        self.writer = SummaryWriter(log_dir=self.log_dir)
        self.step = 0
        
        # Метрики для гистограмм
        self.metric_history = []
        self.curvature_history = []
        
        print(f"TensorBoard логирование в: {self.log_dir}")
        print(f"Запустите: tensorboard --logdir={log_dir}")
        
    def log_metric_field(self, metric_tensor: torch.Tensor, step: int = None):
        """
        Логирование распределения компонент метрического тензора.
        """
        if step is None:
            step = self.step
            
        for mu in range(4):
            for nu in range(4):
                if mu <= nu:  # только независимые компоненты
                    component = metric_tensor[..., mu, nu].flatten().cpu().numpy()
                    self.writer.add_histogram(f"metric/g_{mu}{nu}", component, step)
                    self.writer.add_scalar(f"metric/mean_g_{mu}{nu}", 
                                           np.mean(component), step)
                    self.writer.add_scalar(f"metric/std_g_{mu}{nu}", 
                                           np.std(component), step)
    
    def log_curvature(self, curvature_scalar: torch.Tensor, step: int = None):
        """
        Логирование скалярной кривизны и детектора сингулярностей.
        """
        if step is None:
            step = self.step
            
        curv_flat = curvature_scalar.flatten().cpu().numpy()
        
        self.writer.add_histogram("curvature/R_scalar", curv_flat, step)
        self.writer.add_scalar("curvature/mean_R", np.mean(curv_flat), step)
        self.writer.add_scalar("curvature/max_R", np.max(curv_flat), step)
        self.writer.add_scalar("curvature/singularity_density", 
                               np.sum(curv_flat > 1e50) / len(curv_flat), step)
        
        # Тепловая карта кривизны (2D срез)
        if curvature_scalar.dim() >= 3:
            slice_xy = curvature_scalar[0, :, :, 0].cpu().numpy()  # t=0, z=0
            self.writer.add_image("curvature/heatmap_xy", 
                                  self._normalize_image(slice_xy), step, dataformats='HW')
    
    def log_particle_bundle(self, trajectory_bundle: torch.Tensor, step: int = None):
        """
        Логирование ансамбля частиц (траектории, фазовое пространство).
        """
        if step is None:
            step = self.step
            
        # Финальные позиции
        final_positions = trajectory_bundle[:, -1, 1:4].cpu().numpy()
        
        # 2D проекции для scatter plot
        self.writer.add_scalar("bundle/final_x_mean", np.mean(final_positions[:, 0]), step)
        self.writer.add_scalar("bundle/final_x_std", np.std(final_positions[:, 0]), step)
        self.writer.add_scalar("bundle/final_y_mean", np.mean(final_positions[:, 1]), step)
        self.writer.add_scalar("bundle/final_y_std", np.std(final_positions[:, 1]), step)
        
        # 3D траектории (embedding для projector)
        if len(trajectory_bundle) >= 100:
            sample_trajs = trajectory_bundle[:100, :, 1:4].cpu().numpy()
            for i, traj in enumerate(sample_trajs):
                self.writer.add_scalar(f"trajectory/endpoint_distance_{i}", 
                                      np.linalg.norm(traj[-1] - traj[0]), step)
    
    def log_fractal_dimension(self, fractal_dim: float, step: int = None):
        """
        Логирование фрактальной размерности пучка.
        """
        if step is None:
            step = self.step
            
        self.writer.add_scalar("fractal/dimension", fractal_dim, step)
        
        # Аномалии: D₂ > 3.0 указывает на нелокальность
        if fractal_dim > 3.0:
            self.writer.add_scalar("fractal/anomaly_strength", fractal_dim - 3.0, step)
    
    def log_momentum_dispersion(self, dispersion: torch.Tensor, step: int = None):
        """
        Логирование диссипации импульса.
        """
        if step is None:
            step = self.step
            
        disp_np = dispersion.cpu().numpy()
        
        for i, axis in enumerate(['x', 'y', 'z']):
            self.writer.add_scalar(f"dispersion/axis_{axis}", disp_np[-1, i], step)
            self.writer.add_scalar(f"dispersion/growth_rate_{axis}", 
                                  (disp_np[-1, i] - disp_np[0, i]) / len(disp_np), step)
    
    def log_energy_spectrum(self, energies: np.ndarray, step: int = None):
        """
        Логирование энергетического спектра частиц после взаимодействия с пеной.
        """
        if step is None:
            step = self.step
            
        self.writer.add_histogram("energy/spectrum", energies, step)
        self.writer.add_scalar("energy/mean", np.mean(energies), step)
        self.writer.add_scalar("energy/peak", energies[np.argmax(np.bincount(energies.astype(int)))], step)
        
        # Подгонка степенного закона (для турбулентности)
        bins = np.logspace(np.log10(min(energies)), np.log10(max(energies)), 50)
        hist, bin_edges = np.histogram(energies, bins=bins)
        bin_centers = (bin_edges[:-1] + bin_edges[1:]) / 2
        
        # Степенная аппроксимация dN/dE ~ E^{-α}
        mask = hist > 0
        if np.sum(mask) > 5:
            log_hist = np.log(hist[mask])
            log_energy = np.log(bin_centers[mask])
            
            from scipy.stats import linregress
            slope, _, r_value, _, _ = linregress(log_energy, log_hist)
            
            self.writer.add_scalar("energy/spectral_index", -slope, step)
            self.writer.add_scalar("energy/spectral_fit_r2", r_value**2, step)
    
    def log_virtual_singularities(self, singularities: list, step: int = None):
        """
        Логирование рождённых и коллапсировавших сингулярностей.
        """
        if step is None:
            step = self.step
            
        active = sum(1 for s in singularities if s.is_active)
        collapsed = len(singularities) - active
        
        self.writer.add_scalar("singularities/active", active, step)
        self.writer.add_scalar("singularities/collapsed", collapsed, step)
        self.writer.add_scalar("singularities/total_born", len(singularities), step)
        
        if active > 0:
            lifetimes = [s.lifetime for s in singularities if s.is_active]
            self.writer.add_histogram("singularities/lifetimes", lifetimes, step)
    
    def _normalize_image(self, data: np.ndarray) -> np.ndarray:
        """
        Нормализация изображения для TensorBoard.
        """
        data_min, data_max = data.min(), data.max()
        if data_max - data_min > 0:
            normalized = (data - data_min) / (data_max - data_min)
        else:
            normalized = np.zeros_like(data)
        return normalized
    
    def log_hyperparameters(self, hparams: Dict[str, Any]):
        """
        Логирование гиперпараметров эксперимента.
        """
        self.writer.add_hparams(hparams, metric_dict={})
        
        # Сохраняем конфиг в JSON
        config_path = os.path.join(self.log_dir, "config.json")
        with open(config_path, 'w') as f:
            json.dump(hparams, f, indent=2)
    
    def close(self):
        self.writer.close()


class StochasticQuantumGravitySimulatorWithMonitor(StochasticQuantumGravitySimulator):
    """
    Расширенный симулятор с TensorBoard мониторингом.
    """
    
    def __init__(self, grid_shape=(32,32,32,32), enable_monitoring=True, 
                 experiment_name=None):
        super().__init__(grid_shape)
        
        self.enable_monitoring = enable_monitoring
        if enable_monitoring:
            self.monitor = TensorBoardQuantumMonitor(
                log_dir="runs/quantum_foam",
                experiment_name=experiment_name
            )
        else:
            self.monitor = None
            
        self.global_step = 0
        self.trajectory_cache = []
        
    def run_with_monitoring(self, n_steps=1000, n_particles=100, 
                           particle_type=ParticleType.NEUTRON,
                           log_every=10, save_checkpoint_every=100):
        """
        Основной цикл с реальным временным мониторингом.
        """
        if not self.enable_monitoring:
            print("Мониторинг отключён. Используйте run() без логирования.")
            return
        
        # Логируем гиперпараметры
        hparams = {
            'grid_shape': self.grid_shape,
            'n_steps': n_steps,
            'n_particles': n_particles,
            'particle_type': particle_type.name,
            'log_every': log_every,
            'planck_length': 1.616e-35,
            'planck_time': 5.39e-44,
        }
        self.monitor.log_hyperparameters(hparams)
        
        print(f"Запуск мониторинга на {n_steps} шагов...")
        print("="*60)
        
        for step in range(n_steps):
            # 1. Генерация метрических флуктуаций
            delta_g = self.metric.generate_stochastic_fluctuations()
            
            # 2. Детектирование сингулярностей
            new_singularities = self.detect_singularities(delta_g)
            self.singularities.extend(new_singularities)
            
            # 3. Эволюция существующих
            for s in self.singularities[:]:
                s.evolve(step * 5.39e-44, delta_g)
                if not s.is_active:
                    # Коллапс → рождение запутанной пары
                    pair = EntangledPair(s.pos, self.random_remote_point(s.pos))
                    self.entangled_pairs.append(pair)
                    self.singularities.remove(s)
            
            # 4. Периодическое логирование
            if step % log_every == 0:
                # Логируем метрическое поле
                self.monitor.log_metric_field(delta_g, step)
                
                # Логируем кривизну
                curvature = self.compute_curvature_scalar(delta_g)
                self.monitor.log_curvature(curvature, step)
                
                # Логируем сингулярности
                self.monitor.log_virtual_singularities(self.singularities, step)
                
                # Запускаем ансамблевую симуляцию каждые 100 шагов
                if step % 100 == 0 and step > 0:
                    print(f"\nШаг {step}: Запуск ансамблевого анализа...")
                    
                    # Создаём временный пропагатор с текущей метрикой
                    propagator = HybridPropagator(delta_g.cpu().numpy())
                    
                    # Запускаем ансамбль
                    results = propagator.run_ensemble_simulation(
                        n_particles=n_particles,
                        n_steps=50,  # короткая симуляция для мониторинга
                        particle_type=particle_type,
                        initial_energy_mev=100.0
                    )
                    
                    # Логируем результаты
                    self.monitor.log_particle_bundle(results['trajectories'], step)
                    self.monitor.log_fractal_dimension(results['fractal_dimension'], step)
                    self.monitor.log_momentum_dispersion(results['momentum_dispersion'], step)
                    
                    # Логируем энергетический спектр
                    if particle_type == ParticleType.NEUTRON:
                        # Извлекаем энергии из 4-скоростей
                        velocities = results['trajectories'][:, -1, 5:8].cpu().numpy()
                        energies = 939.565 * (1 / np.sqrt(1 - np.linalg.norm(velocities, axis=1)**2 / 9e16) - 1)
                        self.monitor.log_energy_spectrum(energies, step)
                    
                    print(f"  Фрактальная размерность: D₂ = {results['fractal_dimension']:.4f}")
                    print(f"  Дисперсия импульса: {results['momentum_dispersion'][-1].cpu().numpy()}")
                
                # Логируем шкалу времени
                self.monitor.writer.add_scalar("simulation/planck_time", step * 5.39e-44, step)
                self.monitor.writer.add_scalar("simulation/step", step, step)
            
            # Сохраняем чекпоинт
            if step % save_checkpoint_every == 0 and step > 0:
                self.save_checkpoint(step)
            
            # Прогресс-бар
            if step % (n_steps // 10) == 0:
                progress = step / n_steps * 100
                print(f"Прогресс: {progress:.1f}% (шаг {step}/{n_steps})")
        
        print("\n" + "="*60)
        print("Симуляция завершена!")
        print(f"TensorBoard данные сохранены в: {self.monitor.log_dir}")
        print("Запустите просмотр: tensorboard --logdir=runs/quantum_foam")
        
        self.monitor.close()
        
        return self.collect_final_metrics()
    
    def save_checkpoint(self, step: int):
        """
        Сохранение состояния симуляции.
        """
        checkpoint_dir = os.path.join(self.monitor.log_dir, "checkpoints")
        os.makedirs(checkpoint_dir, exist_ok=True)
        
        checkpoint = {
            'step': step,
            'singularities': [(s.pos, s.lifetime, s.is_active) for s in self.singularities],
            'entangled_pairs': [(p.point_a, p.point_b) for p in self.entangled_pairs],
            'global_step': self.global_step
        }
        
        torch.save(checkpoint, os.path.join(checkpoint_dir, f"checkpoint_step_{step}.pt"))
        print(f"  Чекпоинт сохранён: шаг {step}")
    
    def collect_final_metrics(self) -> dict:
        """
        Сбор итоговых метрик для отчёта.
        """
        return {
            'total_singularities': len(self.singularities),
            'total_entangled_pairs': len(self.entangled_pairs),
            'final_step': self.global_step,
            'monitor_log_dir': self.monitor.log_dir if self.monitor else None
        }

    def add_trajectory_embedding(self, trajectories: torch.Tensor, step: int, 
                                 metadata: Optional[list] = None):
        """
        Добавление траекторий в Projector для 3D визуализации.
        """
        from torch.utils.tensorboard import SummaryWriter
        import tensorflow as tf
        
        # Извлекаем конечные точки и промежуточные позиции

            endpoints,
            metadata=metadata,
            tag=f"trajectory_endpoints_step_{step}",
            global_step=step
        )


# Интерактивный мониторинг в реальном времени
class LiveQuantumMonitor:
    """
    Потоковый мониторинг с обновлением в реальном времени.
    """
    
    def __init__(self, simulator: StochasticQuantumGravitySimulatorWithMonitor):
        self.sim = simulator
        self.monitor = simulator.monitor
        
    def stream_metrics(self, duration_planck_steps: int = 10000):
        """
        Потоковая передача метрик в TensorBoard с высоким разрешением.
        """
        import time
        from tqdm import tqdm
        
        print("Запуск потокового мониторинга квантовой пены...")
        print("="*60)
        
        pbar = tqdm(total=duration_planck_steps)
        
        for step in range(duration_planck_steps):
            start_time = time.time()
            
            # Один шаг симуляции
            delta_g = self.sim.metric.generate_stochastic_fluctuations()
            
            # Логирование с высоким разрешением (каждый шаг)
            self.monitor.log_metric_field(delta_g, step)
            
            # Мониторинг флуктуаций метрики
            trace = np.trace(delta_g.cpu().numpy()[..., :4, :4], axis1=-2, axis2=-1)
            self.monitor.writer.add_scalar("live/mean_trace", np.mean(trace), step)
            self.monitor.writer.add_scalar("live/trace_std", np.std(trace), step)
            
            # Детектор квантовых всплесков
            if np.std(trace) > 0.1 * np.mean(np.abs(trace)):
                self.monitor.writer.add_scalar("live/quantum_burst", 1.0, step)
                print(f"\n⚠️ Квантовый всплеск на шаге {step}!")
            else:
                self.monitor.writer.add_scalar("live/quantum_burst", 0.0, step)
            
            # Частота обновления
            elapsed = time.time() - start_time
            self.monitor.writer.add_scalar("performance/step_time_ms", elapsed * 1000, step)
            
            pbar.update(1)
            
            # Ограничение для интерактивности
            if step % 1000 == 0 and step > 0:
                print(f"\n📊 Статистика после {step} шагов:")
                print(f"  Средняя флуктуация метрики: {np.mean(np.abs(trace)):.2e}")
                print(f"  Максимальная кривизна: {self.sim.compute_curvature_scalar(delta_g).max():.2e}")
        
        pbar.close()
